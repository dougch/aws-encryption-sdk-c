21 /*
22  * This mock materials cache is designed to ease testing the caching CMM
23  * by emulating a cache with a single entry.
24  *
25  * When encryption materials are put into the cache, the cache will
26  * update its state (enc/dec_materials, encryption_context, usage_stats,
27  * last_cache_id) to match the inserted materials.
28  *
29  * When an entry is looked up, the cache will simulate a cache hit if
30  * should_hit is true; last_cache_it will be set to the ID that was queried.
31  * When the entry is queried for usage stats, creation time, or the
32  * actual materials, the appropriate fields from the mock cache will be
33  * returned and/or updated.
34  */
35 struct mock_materials_cache {
36  struct aws_cryptosdk_materials_cache base;
37  struct aws_allocator *alloc;
38 
39  /*
40  * The entry pointer returned from cache vtable methods points to this
41  * entry_marker field. The value of the field is ignored; we just use
42  * this to assert that the correct entry pointer is being passed around.
43  */
44  uint8_t entry_marker;
45  /* True if find_entry should find a cache hit */
46  bool should_hit;
47  /* True if the call should be made to fail */
48  bool should_fail;
49 
50  /*
51  * Encryption materials for the cached entry; updated on put_entry_for_encrypt,
52  * read on get_enc_materials. If NULL, get_enc_materials fails.
53  */
54  struct aws_cryptosdk_enc_materials *enc_materials;
55  /*
56  * Encryption materials for the cached entry; updated on put_entry_for_decrypt,
57  * read on get_dec_materials. If NULL, get_dec_materials fails.
58  */
59  struct aws_cryptosdk_dec_materials *dec_materials;
60  /*
61  * Encryption context for the cached entry; updated on put_entry_for_encrypt,
62  * read on get_enc_materials.
63  * TODO: Does this need to be used for decrypt entries as well?
64  */
65  struct aws_hash_table encryption_context;
66  /*
67  * Usage stats for the cached entry. Set on put_entry_for_encrypt;
68  * read and updated on update_usage_stats
69  */
70  struct aws_cryptosdk_cache_usage_stats usage_stats;
71 
72  /*
73  * Contains the last cache ID passed to any function that takes a cache ID
74  */
75  struct aws_byte_buf last_cache_id;
76 
77  /*
78  * The creation time for the current cache entry. Returned from entry_creation_time.
79  */
80  uint64_t entry_creation_time;
81  /*
82  * The expiration time for the current cache entry. Set by entry_ttl_hint.
83  */
84  uint64_t entry_ttl_hint;
85  /*
86  * True if entry_release has been called with invalidate = true.
87  */
88  bool invalidated;
89  /*
90  * Outstanding reference count for the entry.
91  */
92  size_t entry_refcount;
93 };
